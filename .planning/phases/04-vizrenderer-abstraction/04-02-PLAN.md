```typescript
      throw new Error('sketch error')
    })
    const renderer = new P5VizRenderer(sketchFactory)
```

```typescript
import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest'
import { renderHook } from '@testing-library/react'
import { useRef } from 'react'

// Mock mountVizRenderer to avoid p5/ResizeObserver side effects
const mockRenderer = {
  mount: vi.fn(),
  resize: vi.fn(),
  pause: vi.fn(),
  resume: vi.fn(),
  destroy: vi.fn(),
}
const mockDisconnect = vi.fn()

vi.mock('../visualizers/mountVizRenderer', () => ({
  mountVizRenderer: vi.fn(() => ({
    renderer: mockRenderer,
    disconnect: mockDisconnect,
  })),
}))

import { mountVizRenderer } from '../visualizers/mountVizRenderer'
const mockMountVizRenderer = mountVizRenderer as unknown as ReturnType<typeof vi.fn>

import { useVizRenderer } from '../visualizers/useVizRenderer'
import type { VizRenderer } from '../visualizers/types'

describe('useVizRenderer', () => {
  beforeEach(() => {
    vi.clearAllMocks()
  })

  function renderHookWithSource(source: () => VizRenderer) {
    const containerElement = document.createElement('div')
    return renderHook(() => {
      const containerRef = useRef<HTMLDivElement | null>(containerElement)
      useVizRenderer(containerRef, source, null, null, null)
    })
  }

  it('calls mountVizRenderer on mount', () => {
    const source = vi.fn(() => mockRenderer) as unknown as () => VizRenderer
    renderHookWithSource(source)

    expect(mockMountVizRenderer).toHaveBeenCalledTimes(1)
  })

  it('passes container, source, refs, size, and onError to mountVizRenderer', () => {
    const source = vi.fn(() => mockRenderer) as unknown as () => VizRenderer
    renderHookWithSource(source)

    const [container, passedSource, refs, size, onError] = mockMountVizRenderer.mock.calls[0]
    expect(container).toBeInstanceOf(HTMLDivElement)
    expect(passedSource).toBe(source)
    expect(refs).toHaveProperty('hapStreamRef')
    expect(refs).toHaveProperty('analyserRef')
    expect(refs).toHaveProperty('schedulerRef')
    expect(typeof size.w).toBe('number')
    expect(typeof size.h).toBe('number')
    expect(typeof onError).toBe('function')
  })

  it('calls disconnect and destroy on unmount', () => {
    const source = vi.fn(() => mockRenderer) as unknown as () => VizRenderer
    const { unmount } = renderHookWithSource(source)

    expect(mockDisconnect).not.toHaveBeenCalled()
    expect(mockRenderer.destroy).not.toHaveBeenCalled()

    unmount()

    expect(mockDisconnect).toHaveBeenCalledTimes(1)
    expect(mockRenderer.destroy).toHaveBeenCalledTimes(1)
  })

  it('disconnect is called before destroy on unmount', () => {
    const callOrder: string[] = []
    mockDisconnect.mockImplementation(() => callOrder.push('disconnect'))
    mockRenderer.destroy.mockImplementation(() => callOrder.push('destroy'))

    const source = vi.fn(() => mockRenderer) as unknown as () => VizRenderer
    const { unmount } = renderHookWithSource(source)
    unmount()

    expect(callOrder).toEqual(['disconnect', 'destroy'])
  })
})
```
